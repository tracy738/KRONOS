local ESP = {
    Enabled      = false,
    ShowSkeleton = false,
    ShowBox      = false,
    ShowTracer   = false,
    ShowName     = false,
    ShowHealth   = false,
    ShowDistance = false,
    SkeletonColor= Color3.fromRGB(255,255,255),
    BoxColor     = Color3.fromRGB(255,255,255),
    TracerColor  = Color3.fromRGB(255,255,255),
}

local function newDrawing(class, props)
    local d = Drawing.new(class)
    for k,v in pairs(props) do d[k]=v end
    return d
end

local function createEspEntry(player)
    local e = {
        box         = newDrawing("Square",{Thickness=1,Filled=false,Visible=false,Color=ESP.BoxColor}),
        boxOutline  = newDrawing("Square",{Thickness=3,Filled=false,Visible=false,Color=Color3.new(0,0,0)}),
        name        = newDrawing("Text",  {Size=13,Outline=true,Center=true,Visible=false,Color=Color3.new(1,1,1)}),
        healthBG    = newDrawing("Line",  {Thickness=3,Visible=false,Color=Color3.new(0,0,0)}),
        healthBar   = newDrawing("Line",  {Thickness=1,Visible=false}),
        distance    = newDrawing("Text",  {Size=11,Outline=true,Center=true,Visible=false,Color=Color3.new(1,1,1)}),
        tracer      = newDrawing("Line",  {Thickness=1,Visible=false,Color=ESP.TracerColor}),
        skeleLines  = {},
        boxLines    = {},
    }
    espCache[player] = e
end

local function removeEspEntry(player)
    local e = espCache[player]
    if not e then return end
    for _, d in pairs(e) do
        if typeof(d) == "userdata" then pcall(function() d:Remove() end) end
    end
    for _, ln in ipairs(e.skeleLines) do pcall(function() ln[1]:Remove() end) end
    for _, ln in ipairs(e.boxLines)   do pcall(function() ln:Remove()    end) end
    espCache[player] = nil
end

local function hideAll(e)
    for _, d in pairs(e) do
        if typeof(d) == "userdata" then pcall(function() d.Visible = false end) end
    end
    for _, ln in ipairs(e.skeleLines) do pcall(function() ln[1].Visible = false end) end
    for _, ln in ipairs(e.boxLines)   do pcall(function() ln.Visible = false    end) end
end

RunService.RenderStepped:Connect(function()
    camera = workspace.CurrentCamera
    for player, e in pairs(espCache) do
        local char = player.Character
        -- If ESP is disabled or player has no character, hide everything and skip
        if not ESP.Enabled or not char then 
            hideAll(e) 
            continue 
        end

        local root = char:FindFirstChild("HumanoidRootPart")
        local head = char:FindFirstChild("Head")
        local hum  = char:FindFirstChildOfClass("Humanoid")
        
        -- Basic validation
        if not (root and head and hum and hum.Health > 0) then 
            hideAll(e) 
            continue 
        end

        local screenPos, onScreen = camera:WorldToViewportPoint(root.Position)
        
        -- If player is behind the camera, hide and skip
        if not onScreen then 
            hideAll(e) 
            continue 
        end

        -- Calculate Box Dimensions
        local topPos = camera:WorldToViewportPoint(root.Position + Vector3.new(0, 2.6, 0))
        local botPos = camera:WorldToViewportPoint(root.Position - Vector3.new(0, 3.0, 0))
        local charH  = math.abs(topPos.Y - botPos.Y)
        local charW  = charH * 0.6
        local bPos   = Vector2.new(screenPos.X - charW/2, screenPos.Y - charH * 0.55)
        local bSize  = Vector2.new(charW, charH)

        -- BOX LOGIC
        if ESP.ShowBox then
            e.boxOutline.Size     = bSize
            e.boxOutline.Position = bPos
            e.boxOutline.Visible  = true
            e.box.Size            = bSize
            e.box.Position        = bPos
            e.box.Color           = ESP.BoxColor
            e.box.Visible         = true
        else
            e.box.Visible = false
            e.boxOutline.Visible = false
        end

        -- NAME LOGIC
        if ESP.ShowName then
            e.name.Text     = player.Name
            e.name.Position = Vector2.new(bPos.X + bSize.X/2, bPos.Y - 16)
            e.name.Visible  = true
        else
            e.name.Visible = false
        end

        -- HEALTH LOGIC
        if ESP.ShowHealth then
            local pct = math.clamp(hum.Health / math.max(hum.MaxHealth, 1), 0, 1)
            local x   = bPos.X - 5
            e.healthBG.From    = Vector2.new(x, bPos.Y + bSize.Y)
            e.healthBG.To      = Vector2.new(x, bPos.Y)
            e.healthBG.Visible = true
            e.healthBar.From   = Vector2.new(x, bPos.Y + bSize.Y)
            e.healthBar.To     = Vector2.new(x, bPos.Y + bSize.Y - bSize.Y * pct)
            e.healthBar.Color  = Color3.fromRGB(math.floor(255*(1-pct)), math.floor(255*pct), 0)
            e.healthBar.Visible = true
        else
            e.healthBG.Visible  = false
            e.healthBar.Visible = false
        end

        -- DISTANCE LOGIC
        if ESP.ShowDistance then
            local dist = math.floor((camera.CFrame.Position - root.Position).Magnitude)
            e.distance.Text     = dist.."m"
            e.distance.Position = Vector2.new(bPos.X + bSize.X/2, bPos.Y + bSize.Y + 4)
            e.distance.Visible  = true
        else
            e.distance.Visible = false
        end

        -- TRACER LOGIC
        if ESP.ShowTracer then
            e.tracer.From    = Vector2.new(camera.ViewportSize.X/2, camera.ViewportSize.Y)
            e.tracer.To      = Vector2.new(screenPos.X, screenPos.Y)
            e.tracer.Color   = ESP.TracerColor
            e.tracer.Visible = true
        else
            e.tracer.Visible = false
        end

        -- SKELETON LOGIC
        if ESP.ShowSkeleton then
            if #e.skeleLines == 0 then
                for _, pair in ipairs(bones) do
                    local sl = newDrawing("Line",{Thickness=1,Color=ESP.SkeletonColor,Visible=false})
                    table.insert(e.skeleLines, {sl, pair[1], pair[2]})
                end
            end
            for _, data in ipairs(e.skeleLines) do
                local sl, b1, b2 = data[1], data[2], data[3]
                local p1, p2 = char:FindFirstChild(b1), char:FindFirstChild(b2)
                if p1 and p2 then
                    local s1, v1 = camera:WorldToViewportPoint(p1.Position)
                    local s2, v2 = camera:WorldToViewportPoint(p2.Position)
                    if v1 and v2 then
                        sl.From, sl.To = Vector2.new(s1.X, s1.Y), Vector2.new(s2.X, s2.Y)
                        sl.Visible = true
                    else sl.Visible = false end
                else sl.Visible = false end
            end
        else
            for _, data in ipairs(e.skeleLines) do data[1].Visible = false end
        end
    end
end)

local function tw(o,p,t,s,d)
    local tk=TweenService:Create(o,TweenInfo.new(t or .2,s or Enum.EasingStyle.Quint,d or Enum.EasingDirection.Out),p)
    tk:Play() return tk
end
local function corner(p,r) local c=Instance.new("UICorner") c.CornerRadius=UDim.new(0,r or 8) c.Parent=p return c end

local function grad(p,a,b,r) local g=Instance.new("UIGradient") g.Color=ColorSequence.new(a,b) g.Rotation=r or 90 g.Parent=p return g end
